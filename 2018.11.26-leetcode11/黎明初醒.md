class Solution {
    public int maxArea(int[] height) {
        int i=0,j=height.length-1;
        int V=0,V1;
        while(i<=j) {
        	if(height[i]<=height[j]) {
        		V1 = (j-i)*height[i];
        		i++;
        	}
        	else {
        		V1 = (j-i)*height[j];
        		j--;
        	}
        	if(V1>=V) {
        		V = V1;
        	}
        	
        }
        return V;
    }
}
